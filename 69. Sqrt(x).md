# 69. Sqrt(x)
```
/**
 * 二分搜索 [1, x] 内的整数, 找到最适合的一个
 */
int mySqrt(int x)
{
    int ans = 0;
    unsigned long long mid, left, right, mid_v;
    left = 0;
    right = x - 1; // right = x / 2 + 1 可以缩小搜索范围
    // 把 [1,x] 看作一个数组A, 下标为 0 ~ x-1
    // mid_v = A[mid]
    while (left <= right)
    {
        mid = (left + right) / 2;
        mid_v = mid + 1;
        if (mid_v * mid_v == x)
            return mid_v;
        if (mid_v * mid_v > x)
        {
            right = mid - 1;
        }
        else
        {
            left = mid + 1;
            ans = mid_v;
        }
    }
    return ans;
}
```
