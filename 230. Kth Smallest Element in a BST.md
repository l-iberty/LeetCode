# 230. Kth Smallest Element in a BST
```
// 以中序遍历的非递归算法为框架. 中序遍历取得的顶点是依次递增的,每遍历一个顶点
// k就减1,直到k减为0
int kthSmallest(TreeNode* root, int k)
{
    stack<TreeNode*> s;
    TreeNode* p = root;

    while (p)
    {
        s.push(p);
        p = p->left;
    }
    while (!s.empty())
    {
        p = s.top();
        s.pop();

        if (--k == 0)
            return p->val;

        p = p->right;
        while (p)
        {
            s.push(p);
            p = p->left;
        }
    }
    return -1;
}
```
