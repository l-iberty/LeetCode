# 173. Binary Search Tree Iterator
## Solution 1
将中序序列保存在队列中，每次调用`next()`时直接返回队首元素，空间复杂度为`O(n)`
```
class BSTIterator {
private:
    queue<int> mQueue;
public:
    BSTIterator(TreeNode* root)
    {
        stack<TreeNode*> s;
        TreeNode *p = root;
        
        while (p)
        {
            s.push(p);
            p = p->left;
        }
        
        while (!s.empty())
        {
            p = s.top();
            s.pop();
            mQueue.push(p->val);
            
            p = p->right;
            while (p)
            {
                s.push(p);
                p = p->left;
            }
        }
    }
    
    /** @return the next smallest number */
    int next()
    {
        int n = mQueue.front();
        mQueue.pop();
        return n;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext()
    {
        return !mQueue.empty();
    }
};
```

## Solution 2
在构造函数内不直接生成完整的中序序列，而是每次调用`next()`时才将遍历过程向前推进，这样可以将辅助栈的长度控制在`O(h)`
```
class BSTIterator {
private:
    stack<TreeNode*> mStack;
public:
    BSTIterator(TreeNode* root)
    {
        TreeNode *p = root;
        while (p)
        {
            mStack.push(p);
            p = p->left;
        }
    }
    
    /** @return the next smallest number */
    int next()
    {
        TreeNode *p = mStack.top();
        mStack.pop();
        int n = p->val;
        
        p = p->right;
        while (p)
        {
            mStack.push(p);
            p = p->left;
        }
        
        return n;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext()
    {
        return !mStack.empty();
    }
};
```
