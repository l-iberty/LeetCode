# 494. Target Sum
```
int findTargetSumWays(vector<int>& nums, int S)
{
    int ans = 0;
    helper(ans, nums, 0, 0, S);
    return ans;
}

void helper(int& ans, vector<int>& nums, int i, int current, int S)
{
    if (i >= nums.size())
    {
        if (current == S) ans++;
    }
    else
    {
        helper(ans, nums, i + 1, current + nums[i], S);
        helper(ans, nums, i + 1, current - nums[i], S);
    }
}
```
