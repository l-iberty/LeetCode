# 310. Minimum Height Trees
## (1) Brute force
遍历所有节点，求出以每个节点为根的树高，高度最小者即为所求
```
void printGraph(vector<vector<int>>& graph)
{
	for (int i = 0; i < graph.size(); i++)
	{
		printf("[%d] ", i);
		for (int j = 0; j < graph[i].size(); j++)
		{
			printf("%d ", graph[i][j]);
		}
		printf("\n");
	}
}

// build indirected graph using adjacent table
vector<vector<int>> buildGraph(int n, vector<pair<int, int>>& edges)
{
	vector<vector<int>> graph(n);
	int E = edges.size();
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < E; j++)
		{
			if (edges[j].first == i)
			{
				graph[i].push_back(edges[j].second);
			}
			else if (edges[j].second == i)
			{
				graph[i].push_back(edges[j].first);
			}
		}
	}
	return graph;
}

// 对每个节点进行BFS, 得到以它为根的树高, 高度最小者即为所求
vector<int> findMinHeightTrees(int n, vector<pair<int, int>>& edges)
{
	vector<int> roots;
	vector<vector<int>> graph = buildGraph(n, edges);
	int height, min_height = INT_MAX;
	//printGraph(graph);

	for (int i = 0; i < n; i++)
	{
		queue<int> q;
		vector<bool> visited(n, false);

		q.push(i);
		height = 0;
		while (!q.empty())
		{
			int n = q.size();
			while (n--)
			{
				int u = q.front();
				q.pop();
				visited[u] = true;
				for (int v = 0; v < graph[u].size(); v++)
				{
					if (!visited[graph[u][v]])
						q.push(graph[u][v]);
				}
			}
			height++;
		}
		//printf("h[%d] = %d\n", i, height);
		if (height < min_height)
		{
			roots.clear();
			min_height = height;
			roots.push_back(i);
		}
		else if (height == min_height)
		{
			roots.push_back(i);
		}
	}

	return roots;
}
```

## (2) 剥洋葱
将外层叶子节点(只有一条连接边的节点)一层一层地剥去，最后剩下的1个或2个节点即为所求
```
void printGraph(vector<vector<int>>& graph)

// build indirected graph using adjacent table
vector<vector<int>> buildGraph(int n, vector<pair<int, int>>& edges)
{
	vector<vector<int>> graph(n);
	int e = edges.size();
	for (int i = 0; i < n; i++)
	{
		graph[i].push_back(i);
		for (int j = 0; j < e; j++)
		{
			if (edges[j].first == i)
			{
				graph[i].push_back(edges[j].second);
			}
			else if (edges[j].second == i)
			{
				graph[i].push_back(edges[j].first);
			}
		}
	}
	return graph;
}

// 取回并剔除外层叶子节点(只有一条连接边的节点)
vector<int> retrieveLeaves(vector<vector<int>>& graph)
{
	vector<int> leaves;
	int i;
	for (i = 0; i < graph.size(); i++)
	{
		if (graph[i].size() == 2) // 只有一条连边
		{
			leaves.push_back(graph[i][0]);
		}
	}

	for (i = 0; i < leaves.size(); i++)
	{
		vector<vector<int>>::iterator it1 = graph.begin();
		while (it1 != graph.end())
		{
			if ((*it1)[0] == leaves[i])
			{
				it1 = graph.erase(it1);
			}
			else
			{
				vector<int>::iterator it2 = it1->begin();
				while (it2 != it1->end())
				{
					if (*it2 == leaves[i])
					{
						it2 = it1->erase(it2);
					}
					else
					{
						it2++;
					}
				}
				it1++;
			}
		}
	}
	return leaves;
}

vector<int> findMinHeightTrees(int n, vector<pair<int, int>>& edges)
{
	vector<int> roots;
	if (n == 1)
	{
		roots.push_back(0);
		return roots;
	}

	vector<vector<int>> graph = buildGraph(n, edges);
	//printGraph(graph);
	while (1)
	{
		if (graph.size() <= 2)
		{
			for (int i = 0; i < graph.size(); i++)
			{
				roots.push_back(graph[i][0]);
			}
			break;
		}
		retrieveLeaves(graph); // 剥去一层叶子节点
		//printGraph(graph);
	}
	return roots;
}
```
