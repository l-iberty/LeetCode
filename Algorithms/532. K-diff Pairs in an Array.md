# 532. K-diff Pairs in an Array
```
int findPairs(vector<int>& nums, int k)
{
    if (k < 0) return 0;

    int ans = 0;
    map<int,int> m; // 数字a - 数字a出现的次数

    for (int a : nums)
    {
        if (m.find(a) == m.end())
            m[a] = 1;
        else
            m[a]++;
    }

    map<int,int>::iterator it = m.begin();

    // 如果k=0, 则必须要有重复数字出现.
    // 每遇到一个重复数字a, ans就自增, 并将a删除, 因为不管有几个重复的a,
    // 都只能构成一个数对(a,a)
    if (k == 0)
    {
        for (; it != m.end(); it++)
        {
            if (it->second > 1)
            {
                ans++;
                m.erase(it->first);
            }
        }
        return ans;
    }

    // 如果k>0, 则遍历map, 对于其中的每个数字a, 检查(a+k)或(a-k)是否存在,
    // 并将a删除, 避免重复计数.
    for (; it != m.end(); it++)
    {
        int a = it->first;
        if (m.find(a + k) != m.end()) ans++;
        if (m.find(a - k) != m.end()) ans++;
        m.erase(a);
    }

    return ans;
}
```
