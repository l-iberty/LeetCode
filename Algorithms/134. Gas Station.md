# 134. Gas Station
## #1 逐个尝试
```
bool check(vector<int>& gas, vector<int>& cost, int start)
{
    int tank = 0;
    int i = start, n = gas.size();

    do
    {
        tank = tank + gas[i] - cost[i];
        if (tank < 0)
        {
            return false;
        }
        i = (i + 1) % n;
    } while (i != start);

    return true;
}

int canCompleteCircuit(vector<int>& gas, vector<int>& cost)
{
    for (int i = 0; i < gas.size(); i++)
    {
        if (check(gas, cost, i))
            return i;
    }
    return -1;
}
```
