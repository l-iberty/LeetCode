# 922. Sort Array By Parity II
```
vector<int> sortArrayByParityII(vector<int>& A)
{
    vector<int> ans(A.size());
    int i = 0, j = 1;

    for (int a : A)
    {
        if (a % 2 == 0)
        {
            ans[i] = a;
            i += 2;
        }
        else if (a % 2 == 1)
        {
            ans[j] = a;
            j += 2;
        }
    }
    return ans;
}
```
