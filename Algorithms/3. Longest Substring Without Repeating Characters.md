# 3. Longest Substring Without Repeating Characters
```cpp
  int lengthOfLongestSubstring(string s) {
    int res = 0, l = 0, r = 0;
    unordered_map<char, int> counters;
    while (r < s.size()) {
      int count = ++counters[s[r]];
      if (count == 1) {
        res = max(res, r - l + 1);
      } else {
        // 移动 l 直至窗口内不再用重复字符
        counters[s[l]]--;
        l++;
        while (l < r && counters[s[r]] > 1) {
          counters[s[l]]--;
          l++;
        }
      }
      r++;
    }
    return res;
  }
```
