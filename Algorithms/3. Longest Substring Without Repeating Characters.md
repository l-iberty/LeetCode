# 3. Longest Substring Without Repeating Characters
```cpp
  int lengthOfLongestSubstring(string s) {
    int res = 0, l = 0, r = 0;
    unordered_map<char, int> counters;

    while (r < s.size()) {
      int count = counters[s[r]]++;
      if (count == 0) {
        res = max(res, r - l + 1);
      } else {
        // 出现了重复字符, 需要移动 l 直到窗口内不再有重复字符
        while (l < r) {
          char c = s[l++];
          counters[c]--;
          if (counters[c] == 1) {
            // 重复字符就是此时的c, c就是s[r]
            break;
          }
        }
      }
      r++;
    }
    return res;
  }
```
