# 27. Remove Element
```
int removeElement(int* nums, int numsSize, int val) {
    int ans = 0;
    int i, j, k;
    for (i = 0; i < numsSize; i++)
    {
        if (nums[i] != val)
        {
            ans++;
        }
        else
        {
            for (j = i + 1; j < numsSize; j++)
            {
                if (nums[j] != val)
                {
                    ans++;
                    break;
                }
            }
            if (j < numsSize)
            {
                nums[i] = nums[j];
                nums[j] = val;
            }
        }
    }
    return ans;
}
```
