# 23. Merge k Sorted Lists
```cpp
 public:
  ListNode* mergeKLists(vector<ListNode*>& lists) {
    ListNode *res = nullptr;
    for (ListNode *list : lists) {
      res = merge(res, list);
    }
    return res;
  }

  ListNode *merge(ListNode *L1, ListNode *L2) {
    ListNode head;
    ListNode *p1 = L1, *p2 = L2, *p = &head;
    while (p1 && p2) {
      if (p1->val < p2->val) {
        p->next = p1;
        p1 = p1->next;
      } else {
        p->next = p2;
        p2 = p2->next;
      }
      p = p->next;
    }
    while (p1) {
      p->next = p1;
      p1 = p1->next;
      p = p->next;
    }
    while (p2) {
      p->next = p2;
      p2 = p2->next;
      p = p->next;
    }
    return head.next;
  }
```
