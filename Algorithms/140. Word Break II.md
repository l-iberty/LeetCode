# 140. Word Break II

利用了[208. Implement Trie (Prefix Tree)](208.%20Implement%20Trie%20(Prefix%20Tree).md)，当然也可以使用 STL 容器。

class Trie {
public:
  Trie() { memset(ptrs_, 0, sizeof(ptrs_)); }

  void insert(const string &word) {
    Trie *node = this;
    for (int i = 0; i < word.size(); i++) {
      int idx = word[i] - 'a';
      if (node != nullptr) {
        if (node->ptrs_[idx] == nullptr) {
          node->ptrs_[idx] = new Trie;
        }
        node = node->ptrs_[idx];
        if (i == word.size() - 1) {
          node->ended_ = true;
          return;
        }
      }
    }
  }

  bool search(const string &word) {
    Trie *node = this;
    for (int i = 0; i < word.size(); i++) {
      int idx = word[i] - 'a';
      if (node == nullptr) {
        return false;
      }
      node = node->ptrs_[idx];
      if (node && node->ended_ && i == word.size() - 1) {
        return true;
      }
    }
    return false;
  }

  bool startsWith(const string &prefix) {
    Trie *node = this;
    for (int i = 0; i < prefix.size(); i++) {
      int idx = prefix[i] - 'a';
      if (node == nullptr) {
        return false;
      }
      if (node->ptrs_[idx] && i == prefix.size() - 1) {
        return true;
      }
      node = node->ptrs_[idx];
    }
    return false;
  }

private:
  Trie *ptrs_[26];
  bool ended_{false};
};

class Solution {
public:
  vector<string> wordBreak(string s, vector<string> &wordDict) {
    // 如果 s 里含有 wordDict 里没有的字符就返回空列表, 这样可以减少一些循环
    unordered_set<char> charDict;
    for (string &w : wordDict) {
      for (char c : w) {
        charDict.insert(c);
      }
    }
    for (char c : s) {
      if (charDict.find(c) == charDict.end()) {
        return {};
      }
    }

    for (string &w : wordDict) {
      trie_.insert(w);
    }

    vector<string> res;
    string subres;
    helper(s, 0, subres, res);
    return res;
  }

  void helper(string &s, int start, string &subres, vector<string> &res) {
    if (start == s.size()) {
      res.push_back(subres);
      return;
    }

    for (int i = start; i < s.size(); i++) {
      string word = s.substr(start, i - start + 1);
      if (trie_.search(word)) {
        int oldSize = subres.size();
        if (!subres.empty()) {
          subres.push_back(' ');
        }
        subres.append(word);
        helper(s, i + 1, subres, res);
        subres = subres.substr(0, oldSize);
      }
    }
  }

private:
  Trie trie_;
};
