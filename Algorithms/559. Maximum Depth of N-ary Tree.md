# 559. Maximum Depth of N-ary Tree
```
/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> children;

    Node() {}

    Node(int _val, vector<Node*> _children) {
        val = _val;
        children = _children;
    }
};
*/
int maxDepth(Node* root) 
{
    if (root == NULL) return 0;

    queue<Node*> q;
    int depth = 0;
    q.push(root);
    while (!q.empty())
    {
        int size = q.size();
        while (size-- > 0)
        {
            Node* t = q.front();
            q.pop();
            for (int i = 0; t && i < t->children.size(); i++)
            {
                q.push(t->children[i]);
            }
        }
        depth++;
    }
    return depth;
}
```
