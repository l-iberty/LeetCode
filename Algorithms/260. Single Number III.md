# 260. Single Number III
```
// Hash Table
vector<int> singleNumber(vector<int>& nums)
{
    vector<int> ans;
    map<int, int> m;

    for (int i : nums)
    {
        if (m.find(i) == m.end()) // not found
        {
            m[i] = i;
        }
        else // found
        {
            m.erase(i);
        }
    }

    map<int, int>::iterator it = m.begin();
    for (; it != m.end(); it++)
    {
        ans.push_back(it->first);
    }

    return ans;
}```
