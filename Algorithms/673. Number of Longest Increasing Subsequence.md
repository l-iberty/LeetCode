# 673. Number of Longest Increasing Subsequence
不甚理解
```cpp
  int findNumberOfLIS(vector<int>& nums) {
    vector<int> lengths(nums.size(), 1); // lengths[i] = 以 nums[i] 结尾的 LIS 的长度
    vector<int> counts(nums.size(), 1); // counts[i] = 以 nums[i] 结尾的 LIS 的个数

    int maxlen = 1;
    for (int i = 1; i < nums.size(); i++) {
      for (int j = 0; j < i; j++) {
        if (nums[i] > nums[j]) {
          if (lengths[j] + 1 > lengths[i]) {
            lengths[i] = lengths[j] + 1;
            counts[i] = counts[j];
          } else if (lengths[j] + 1 == lengths[i]) {
            counts[i] += counts[j];
          }
        }
      }
      maxlen = max(lengths[i], maxlen);
    }

    int res = 0;
    for (int i = 0; i < nums.size(); i++) {
      if (lengths[i] == maxlen) {
        res += counts[i];
      }
    }
    return res;
  }
```
