# 713. Subarray Product Less Than K
```
// 维持一个滑动窗口, 窗口内数字的乘积刚好小于k
int numSubarrayProductLessThanK(vector<int>& nums, int k)
{
    if (k <= 1) return 0;

    int prod = 1, left = 0, ans = 0;
    for (int right = 0; right < nums.size(); right++)
    {
        prod *= nums[right];
        while (prod >= k) prod /= nums[left++];
        ans += right - left + 1;
    }

    return ans;
}
```
