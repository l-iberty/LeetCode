# 5. Longest Palindromic Substring
```
string longestPalindrome(string s)
{
    string ans;
    int left = 0, right, l, r, len, max_len = 1;

    int N = s.length();

    if (N == 0) return ans;

    // 回文子串长度为奇数, 如"aba"
    for (int i = 0; i < N; i++)
    {
        l = i - 1;
        r = i + 1;
        while (l >= 0 && r < N)
        {
            if (s[l] == s[r])
            {
                len = r - l + 1;
                if (len > max_len)
                {
                    max_len = len;
                    left = l;
                    right = r;
                }
            }
            else
            {
                break;
            }
            l--;
            r++;
        }
    }

    // 回文子串长度为偶数, 如"abba"
    for (int i = 0; i < N; i++)
    {
        l = i;
        r = i + 1;
        while (l >= 0 && r < N)
        {
            if (s[l] == s[r])
            {
                len = r - l + 1;
                if (len > max_len)
                {
                    max_len = len;
                    left = l;
                    right = r;
                }
            }
            else
            {
                break;
            }
            l--;
            r++;
        }
    }

    ans = s.substr(left, max_len);
    return ans;
}
```
