# 234. Palindrome Linked List
```
bool isPalindrome(ListNode* head)
{
    stack<int> s;
    ListNode* p;

    for (p = head; p; p = p->next)
    {
        s.push(p->val);
    }

    for (p = head; p && !s.empty(); p = p->next, s.pop())
    {
        if (s.top() != p->val)
        {
            return false;
        }
    }
    return true;
}
```
