# 86. Partition List
```
ListNode* partition(ListNode* head, int x)
{
    if (head == NULL) return NULL;

    ListNode *p, *head1, *head2, *p1, *p2;

    head1 = new ListNode(0);
    head2 = new ListNode(0);
    p = head;
    p1 = head1;
    p2 = head2;

    while (p)
    {
        if (p->val < x)
        {
            p1->next = p;
            p1 = p;
        }
        else
        {
            p2->next = p;
            p2 = p;
        }
        p = p->next;
        p1->next = NULL;
        p2->next = NULL;
    }

    if (p1->val < x)
        p1->next = head2->next;
    else
        p2->next = head1->next;

    p1 = head1->next;
    p2 = head2->next;

    if (p1 == NULL) return p2;
    if (p2 == NULL) return p1;

    return (p1->val < p2->val) ? p1 : p2;
}
```
