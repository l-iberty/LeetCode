# 931. Minimum Falling Path Sum

![](images/931. Minimum Falling Path Sum.png)

***

```cpp
    int minFallingPathSum(vector<vector<int>>& matrix) {
      int n = matrix.size();
      for (int i = 1; i < n; i++) {
        for (int j = 0; j < n; j++) {
          int x = matrix[i-1][j];
          if (j > 0) {
            x = min(x, matrix[i-1][j-1]);
          }
          if (j < n - 1) {
            x = min(x, matrix[i-1][j+1]);
          }
          matrix[i][j] += x;
        }
      }
      
      int res = INT32_MAX;
      for (int i = 0; i < n; i++) {
        res = min(res, matrix[n-1][i]);
      }
      return res;
    }
```
