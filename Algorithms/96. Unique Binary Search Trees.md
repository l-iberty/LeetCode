# 96. Unique Binary Search Trees
```
// 卡特兰数: C[n] = C[i] * C[n-i-1], 0 <= i <= n-1
// eg. C[3] = C[0] * C[2] + C[1] * C[1] + C[2] * C[0]
int numTrees(int n)
{
    if (n == 0 || n == 1) return 1;

    vector<int> ans(n + 1, 0);
    int i, j, k;
    ans[0] = ans[1] = 1;
    for (i = 2; i <= n; i++)
    {
        for (j = 0, k = i - 1; j < i; j++, k--)
        {
            ans[i] += ans[j] * ans[k];
        }
    }
    return ans[n];
}
```
