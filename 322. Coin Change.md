# 322. Coin Change
## Approach #3 (Dynamic programming - Bottom up)
```
int coinChange(vector<int>& coins, int amount)
{
    vector<int> dp(amount + 1);
    dp[0] = 0;
    for (int i = 1; i <= amount; i++)
    {
        int min_val = INT_MAX;
        for (int c : coins)
        {
            if (i - c >= 0 && dp[i - c] >= 0)
                min_val = min(min_val, dp[i - c]);
        }
        if (min_val != INT_MAX)
            dp[i] = min_val + 1;
        else
            dp[i] = -1;
    }

    return dp[amount];
}
```
