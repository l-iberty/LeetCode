# 70. Climbing Stairs
```
void dfs(int& ans, vector<int>& sub_ans, int target)
{
    if (target == 0)
    {
        ans++;
    }
    else if (target > 0)
    {
        sub_ans.push_back(1);
        dfs(ans, sub_ans, target - 1);
        sub_ans.pop_back();

        sub_ans.push_back(2);
        dfs(ans, sub_ans, target - 2);
        sub_ans.pop_back();
    }
}

// 动态规划:
// s[n] = 上n个台阶的方法种数, s[n] = s[n-2] + s[n-1]
// 通过找规律得出递推关系式, 等价于求Fib数列的第n项
int climbStairs(int n)
{
    if (n == 1) return 1;
    if (n == 2) return 2;

    int prev1 = 1, prev2 = 2, current = 0;
    for (int i = 3; i <= n; i++)
    {
        current = prev1 + prev2;
        prev1 = prev2;
        prev2 = current;
    }
    return current;
}
```
