# 590. N-ary Tree Postorder Traversal
```
vector<int> postorder(Node* root)
{
    vector<int> ans;
    if (root == NULL)
    {
        return ans;
    }

    stack<Node*> s;
    s.push(root);
    while (!s.empty())
    {
        Node* p = s.top();
        s.pop();
        ans.insert(ans.begin(), p->val);

        for (Node* child : p->children)
        {
            s.push(child);
        }
    }
    return ans;
}
```
