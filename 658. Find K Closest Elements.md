# 658. Find K Closest Elements
```
// 寻找最接近x的元素下标
int binSearch(vector<int>& arr, int left, int right, int x)
{
    if (x <= arr[left]) return left;
    if (x >= arr[right]) return right;

    while (left <= right)
    {
        int mid = (left + right) / 2;
        if (x > arr[mid])
            left = mid + 1;
        else if (x < arr[mid])
            right = mid - 1;
        else
            return mid;
    }
    if (abs(arr[right] - x) < abs(arr[right + 1] - x))
        return right;
    else
        return right + 1;
}

// 找到k个最接近x的元素
vector<int> findClosestElements(vector<int>& arr, int k, int x)
{
    vector<int> ans;
    int N = arr.size();
    int i, j;
    int pos = binSearch(arr, 0, N - 1, x);

    if (pos == 0)
    {
        for (i = 0; i < k; i++) ans.push_back(arr[i]);
    }
    else if (pos == N - 1)
    {
        for (i = 0; i < k; i++) ans.push_back(arr[N - k + i]);
    }
    else
    {
        i = pos, j = pos + 1;
        while (i > 0 && j < N && j - i < k)
        {
            int d1 = abs(arr[i - 1] - x);
            int d2 = abs(arr[j] - x);
            if (d1 <= d2) i--;
            else j++;
        }
        if (i == 0)
        {
            while (k - j > 0) j++;
        }
        else if (j == N)
        {
            while (j - i < k) i--;
        }

        while (i < j)
        {
            ans.push_back(arr[i++]);
        }
    }

    return ans;
}
```
