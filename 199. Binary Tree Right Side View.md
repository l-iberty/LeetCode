# 199. Binary Tree Right Side View
```
vector<int> rightSideView(TreeNode* root) 
{
    vector<int> ans;
    if (root == NULL) { return ans; }

    queue<TreeNode*> q;
    q.push(root);
    while (!q.empty())
    {
        int n = q.size();
        ans.push_back(q.back()->val);
        while (n-- > 0)
        {
            TreeNode* t = q.front();
            q.pop();
            if (t->left)
            {
                q.push(t->left);
            }
            if (t->right)
            {
                q.push(t->right);
            }
        }
    }
    return ans;
}
```
