# 92. Reverse Linked List II
```
ListNode* reverseBetween(ListNode* head, int m, int n)
{
	ListNode *prev, *current, *next, *p, *tail;

	if (head == NULL || head->next == NULL || m == n) return head;

	current = p = tail = head;
	for (int i = 1; i < m; i++)
	{
		p = current;
		current = current->next;
		tail = current;
	}

	prev = NULL;
	next = current->next;
	for (; n > m; n--)
	{
		current->next = prev;
		prev = current;
		current = next;
		next = current->next;
	}
	current->next = prev;
	tail->next = next;
	
	if (m == 1)
	{
		head = current;
	}
	else
	{
		p->next = current;
	}
	
	return head;
}
```
